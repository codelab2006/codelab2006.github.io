作者：黄永健日期：2020-03-07
let helloWorld = 'Hello World'; // string type
const HELLO_WORLD = 'Hello World'; // 'Hello World' type

字符串文字类型

type Easing = 'ease-in' | 'ease-out' | 'ease-in-out';
function animate(easing: Easing) {}
animate('ease-in');
animate('ease-out');
animate('ease-in-out');
animate('xxx'); // Error
